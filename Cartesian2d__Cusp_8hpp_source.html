117 inline void Cusp::setStrainPtr(const T* arg)
118 {

120  std::copy(arg, arg + 4, m_Eps.begin());
121 
122  std::array<double, 4> Epsd;
123  double epsm = GT::Hydrostatic_deviatoric(&m_Eps[0], &Epsd[0]);
124  double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
125  m_yield.set_x(epsd);
126 
127  m_Sig[0] = m_Sig[3] = m_K * epsm;
128

137  m_Sig[0] += g * Epsd[0];
138  m_Sig[1] = g * Epsd[1];
139  m_Sig[2] = g * Epsd[2];
