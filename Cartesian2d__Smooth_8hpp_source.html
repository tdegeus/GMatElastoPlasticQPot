118 template <class T>
119 inline void Smooth::setStrainPtr(const T* arg)
120 {
121  namespace GT = GMatTensor::Cartesian2d::pointer;
122  std::copy(arg, arg + 4, m_Eps.begin());
123 
124  std::array<double, 4> Epsd;
125  double epsm = GT::Hydrostatic_deviatoric(&m_Eps[0], &Epsd[0]);
126  double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
127  m_yield.set_x(epsd);
128 
129  m_Sig[0] = m_Sig[3] = m_K * epsm;
130 
131  if (epsd <= 0.0) {
132  m_Sig[1] = m_Sig[2] = 0.0;
133  return;
134  }
135 
136  double eps_min = 0.5 * (m_yield.yright() + m_yield.yleft());
137  double deps_y = 0.5 * (m_yield.yright() - m_yield.yleft());
138 
139  double g = (m_G / epsd) * (deps_y / M_PI) * sin(M_PI / deps_y * (epsd - eps_min));
140  m_Sig[0] += g * Epsd[0];
141  m_Sig[1] = g * Epsd[1];
142  m_Sig[2] = g * Epsd[2];
143  m_Sig[3] += g * Epsd[3];
144 }
